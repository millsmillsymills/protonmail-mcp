# Test plan implementation

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Raise the protonmail-mcp project's test maturity by adding a three-layer test pyramid (unit / harness-integration / protocol-contract), `golangci-lint` and `govulncheck` gates, and four fuzz targets — covering all 23 MCP tools against the in-process `go-proton-api` dev server with no real Proton account in CI.

**Architecture:** Three test layers with hard boundaries. Unit tests cover pure functions (DTO converters, helpers, error mapper) without booting the harness — these are fuzz and property-test targets. Harness-integration tests boot an in-process `go-proton-api` dev server through the existing `internal/testharness` rig and exercise every tool via the MCP transport; recorded fixtures (replayed by an `httptest.Server` interceptor) cover the few endpoints whose dev-server semantics diverge from production. Protocol-contract tests register tools against a bare `mcp.Server`, marshal the catalog deterministically, and compare to a checked-in golden file — turning every description or schema change into a visible PR diff.

**Tech Stack:** Go 1.26.1, `go-proton-api` (dev server), `modelcontextprotocol/go-sdk`, `gopenpgp/v2`, `zalando/go-keyring` (mock mode in tests), `santhosh-tekuri/jsonschema/v5` (new — JSON Schema validation), `golangci-lint`, `govulncheck`.

**Reference spec:** `docs/superpowers/specs/2026-05-03-test-plan-design.md`.

### Task 8: Property test for `filterSensitiveHeaders`

**Files:**
- Create: `internal/tools/headers_property_test.go`

- [ ] **Step 1: Write the test**

```go
package tools

import (
	"math/rand/v2"
	"strings"
	"testing"
)

// Deny-list — must match sensitiveHeaders in messages.go.
var denyList = []string{"bcc", "x-originating-ip", "x-original-sender-ip", "x-original-sender", "x-real-ip"}

func TestFilterSensitiveHeaders_NoDenyListLeaksAcrossCasingsAndWhitespace(t *testing.T) {
	rng := rand.New(rand.NewPCG(0xC0DE, 0xFEED))
	const iterations = 1000

	for i := 0; i < iterations; i++ {
		in := map[string][]string{}
		// Always include each denied header in a randomly mangled form.
		for _, name := range denyList {
			in[mangle(name, rng)] = []string{"VALUE"}
		}
		// Add some safe noise.
		for _, safe := range []string{"Authentication-Results", "DKIM-Signature", "Received"} {
			in[mangle(safe, rng)] = []string{"x"}
		}

		got := filterSensitiveHeaders(in)

		for k := range got {
			lower := strings.ToLower(strings.TrimSpace(k))
			for _, denied := range denyList {
				if lower == denied {
					t.Fatalf("iter %d: denied header leaked: %q", i, k)
				}
			}
		}
	}
}

func mangle(name string, rng *rand.Rand) string {
	var b strings.Builder
	for _, r := range name {
		if rng.IntN(2) == 0 {
			b.WriteRune(toUpper(r))
		} else {
			b.WriteRune(r)
		}
	}
	// filterSensitiveHeaders normalizes only via strings.ToLower; whitespace
	// padding should NOT match. We deliberately do NOT pad here, because the
	// production behavior is exact-match-after-lowercase. If you want to
	// extend to whitespace tolerance, change sensitiveHeaders lookup first.
	return b.String()
}

func toUpper(r rune) rune {
	if r >= 'a' && r <= 'z' {
		return r - 32
	}
	return r
}
```

func copyResponse(w http.ResponseWriter, resp *http.Response) {
	for k, vs := range resp.Header {
		for _, v := range vs {
			w.Header().Add(k, v)
		}
	}
	w.WriteHeader(resp.StatusCode)
	if resp.Body != nil {
		defer resp.Body.Close()
		_, _ = io.Copy(w, resp.Body)
	}
}

func proxyTo(w http.ResponseWriter, r *http.Request, target string) {
	u, _ := url.Parse(target)
	r2 := r.Clone(r.Context())
	r2.URL.Scheme = u.Scheme
	r2.URL.Host = u.Host
	r2.RequestURI = ""
	resp, err := http.DefaultClient.Do(r2)
	if err != nil {
		http.Error(w, err.Error(), http.StatusBadGateway)
		return
	}
	copyResponse(w, resp)
}

package testharness

import (
	"bytes"
	"encoding/json"
	"io"
	"net/http"
	"os"
	"strings"
)

// Fixture is one recorded request/response pair.
type Fixture struct {
	Request struct {
		Method string         `json:"method"`
		Path   string         `json:"path"`
		Body   map[string]any `json:"body,omitempty"`
	} `json:"request"`
	Response struct {
		Status int            `json:"status"`
		Body   map[string]any `json:"body"`
	} `json:"response"`
}

// LoadFixtures parses each path and returns the loaded fixtures.
func LoadFixtures(paths ...string) ([]Fixture, error) {
	out := make([]Fixture, 0, len(paths))
	for _, p := range paths {
		b, err := os.ReadFile(p)
		if err != nil {
			return nil, err
		}
		var f Fixture
		if err := json.Unmarshal(b, &f); err != nil {
			return nil, err
		}
		out = append(out, f)
	}
	return out, nil
}

// WithFixture installs an interceptor that matches each incoming request to
// the first fixture whose method+path equals the request's, then serves its
// canned response. A request with no match falls through to the dev server.
func WithFixture(fixtures ...Fixture) Option {
	return WithInterceptor(func(r *http.Request) *http.Response {
		for _, f := range fixtures {
			if !strings.EqualFold(f.Request.Method, r.Method) {
				continue
			}
			if f.Request.Path != r.URL.Path {
				continue
			}
			body, _ := json.Marshal(f.Response.Body)
			return &http.Response{
				StatusCode: f.Response.Status,
				Header:     http.Header{"Content-Type": []string{"application/json"}},
				Body:       io.NopCloser(bytes.NewReader(body)),
			}
		}
		return nil
	})
}
```

#!/usr/bin/env bash
set -euo pipefail

# Captures one fixture pair (request + canned response) from a live harness
# run, scrubs sensitive headers, and writes to
# internal/protonraw/testdata/<endpoint>/<case>.json.
#
# Usage: scripts/capture-fixture.sh <endpoint-dir> <case-name>
# Example: scripts/capture-fixture.sh domains list_one_verified

if [[ $# -ne 2 ]]; then
  echo "usage: $0 <endpoint-dir> <case-name>" >&2
  exit 2
fi

ENDPOINT="$1"
CASE="$2"
OUT="internal/protonraw/testdata/${ENDPOINT}/${CASE}.json"

mkdir -p "$(dirname "$OUT")"

# Capture is operator-driven: re-run the harness with PROTONMAIL_MCP_TRACE=1
# and copy the request/response pair from the trace log into $OUT, then run
# this script to scrub. (A future iteration can fully automate.)
if [[ ! -f "$OUT" ]]; then
  echo "create $OUT manually first; see internal/protonraw/testdata/domains/list_one_verified.json for shape" >&2
  exit 3
fi

# Scrub bearer tokens and Proton UID values.
TMP=$(mktemp)
sed -E '
  s/("Authorization":[[:space:]]*)"Bearer [^"]+"/\1"<redacted-token>"/g;
  s/("x-pm-uid":[[:space:]]*)"[a-f0-9]{20,}"/\1"<redacted-uid>"/g;
  s/[A-Za-z0-9]{40,}/<redacted-blob>/g
' "$OUT" > "$TMP"
mv "$TMP" "$OUT"

echo "scrubbed: $OUT"
```
